Create `apps/web/src/hooks/useUsage.ts` following the exact pattern from RESEARCH.md Pattern 5 and the established useFetch ref pattern from Phase 3.

```typescript
import { useState, useEffect, useCallback } from 'react';
import { useFetch } from './useFetch';

export interface UsageData {
  count: number;
  limit: number;
  plan: 'free' | 'paid';
}

export function useUsage() {
  const fetchApi = useFetch();
  const [usage, setUsage] = useState<UsageData | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const refresh = useCallback(async () => {
    try {
      setLoading(true);
      const res = await fetchApi('/api/usage');
      if (res.ok) {
        const data = (await res.json()) as UsageData;
        setUsage(data);
        setError(null);
      } else {
        setError('Could not load usage');
      }
    } catch {
      setError('Could not load usage');
    } finally {
      setLoading(false);
    }
  }, [fetchApi]);

  useEffect(() => {
    refresh();
  }, [refresh]);

  return { usage, loading, error, refresh };
}
```

Key design decisions per context:
- `refresh` is exposed so MatchPage can call it after a successful match run (D-09).
- `fetchApi` from `useFetch` is stable (empty deps via ref pattern) so `refresh` via `useCallback([fetchApi])` is also stable — no infinite render loops.
- Returns `usage: UsageData | null` — null during initial load, populated after first fetch.
- `plan` field typed as `'free' | 'paid'` matching the GET /api/usage response contract (D-02).
